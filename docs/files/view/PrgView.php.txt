namespace view;

/**
 * Class PrgView
 * @package view
 */
class PrgView
{
    /**
     * @return bool True if server request is a POST
     */
    public function isPost() {
        if($_POST)
            return true;
        return false;
    }

    /**
     * Reloads the page with GET
     */
    public function reloadPage() {
        header("Location: " . $_SERVER['REQUEST_URI']);
    }
}
